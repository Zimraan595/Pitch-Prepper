# Core web framework
Flask>=3.0

# Speech-to-text with accurate word-level timestamps (forced alignment).
# WhisperX is the default backend; it pulls in faster-whisper + alignment models.
whisperx>=3.1

# Fallback transcription backend (used if WhisperX is unavailable, or set
# TRANSCRIBE_BACKEND=whisper).
openai-whisper>=20231117

# Audio feature analysis (pitch, RMS volume, silence/pause detection)
librosa>=0.10
numpy>=1.24
soundfile>=0.12

# User accounts & leaderboard storage (MongoDB). Requires a running MongoDB
# server; set MONGO_URI (default mongodb://localhost:27017). Optional — the app
# runs without it, just without sign-in/leaderboard.
pymongo>=4.6

# LLM-based content/structure analysis runs on a LOCAL model via Ollama.
# Install Ollama (https://ollama.com) and run: `ollama pull llama3.1`.
# No Python package and no API key are required — the app calls Ollama's local
# HTTP API directly using the standard library.

# NOTE:
# - openai-whisper requires PyTorch (installed automatically as a dependency)
#   and the `ffmpeg` system binary must be available on PATH.
# - If Ollama isn't running, content analysis falls back to a built-in
